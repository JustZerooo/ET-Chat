  139.      * Handler function for the start of a tag

  145.     private function StartElement($parser, $name, $attrs = array())
  146.     {
  147.         //Make the name of the tag lower case
  148.         $name = strtolower($name);
  149.         
  150.         //Check to see if tag is root-level
  151.         if (count($this->stack) == 0) 
  152.         {

  165.             $parent->AddChild($name, $attrs, count($this->stack), $this->cleanTagNames);

  208. /**
  209.  * XML Tag Object (php5)
  210.  * 
  211.  * This object stores all of the direct children of itself in the $children array. They are also stored by
  212.  * type as arrays. So, if, for example, this tag had 2 <font> tags as children, there would be a class member
  213.  * called $font created as an array. $font[0] would be the first font tag, and $font[1] would be the second.

  216.  *
  217.  * To loop through all of the direct children of a specific tag for this object, it is probably easier

  290.     /**
  291.      * Adds a direct child to this object

  296.      * @param bool $cleanTagName 
  297.      */
  298.     public function AddChild($name, $attrs, $parents, $cleanTagName = true)
  299.     {    
  300.         //If the tag is a reserved name, output an error
  301.         if(in_array($name, array('tagChildren', 'tagAttrs', 'tagParents', 'tagData', 'tagName')))
  302.         {
  303.             trigger_error('You have used a reserved name as the name of an XML tag. Please consult the documentation (http://www.criticaldevelopment.net/xml/) and rename the tag named "'.$name.'" to something other than a reserved name.', E_USER_ERROR);
  304.  
  305.             return;
  306.         }
  307.  
  308.         //Create the child object itself
  309.         $child = new XMLTag($name, $attrs, $parents);
  310.  
  311.         //If the cleanTagName feature is on, replace colons and dashes with underscores
  312.         if($cleanTagName)
  313.             $name = str_replace(array(':', '-'), '_', $name);
  314.         
  315.         //Toss up a notice if someone's trying to to use a colon or dash in a tag name
  316.         elseif(strstr($name, ':') || strstr($name, '-'))
  317.             trigger_error('Your tag named "'.$name.'" contains either a dash or a colon. Neither of these characters are friendly with PHP variable names, and, as such, you may have difficulty accessing them. You might want to think about enabling the cleanTagName feature (pass true as the second argument of the XMLParser constructor). For more details, see http://www.criticaldevelopment.net/xml/', E_USER_NOTICE);

  319.         //If there is no array already set for the tag name being added, 
  320.         //create an empty array for it
  321.         if(!isset($this->$name))
  322.             $this->$name = array();
  323.         
  324.         //Add the reference of it to the end of an array member named for the tag's name
  325.         $this->{$name}[] = &$child;
  326.         
  327.         //Add the reference to the children array member
  328.         $this->tagChildren[] = &$child;
  329.         
  330.         //Return a reference to this object for the stack
  331.         return $this;
  332.     }

  334.     /**
  335.      * Returns the string of the XML document which would be generated from this object
  336.      * 
  337.      * This function works recursively, so it gets the XML of itself and all of its children, which
  338.      * in turn gets the XML of all their children, which in turn gets the XML of all thier children,
  339.      * and so on. So, if you call GetXML from the document root object, it will return a string for
  340.      * the XML of the entire document.
  341.      * 
  342.      * This function does not, however, return a DTD or an XML version/encoding tag. That should be
  343.      * handled by XMLParser::GetXML()
  344.      *
  345.      * @return string 
  346.      */
  347.     public function GetXML()
  348.     {
  349.         //Start a new line, indent by the number indicated in $this->parents, add a <, and add the name of the tag
  350.         $out = "\n".str_repeat("\t", $this->tagParents).'<'.$this->tagName;
  351.  
  352.         //For each attribute, add attr="value"
  353.         foreach($this->tagAttrs as $attr => $value)
  354.             $out .= ' '.$attr.'="'.$value.'"';
  355.         
  356.         //If there are no children and it contains no data, end it off with a />
  357.         if(empty($this->tagChildren) && empty($this->tagData))
  358.             $out .= " />";
  359.         
  360.         //Otherwise...
  361.         else
  362.         {    
  363.             //If there are children
  364.             if(!empty($this->tagChildren))
  365.             {
  366.                 //Close off the start tag
  367.                 $out .= '>';
  368.                 
  369.                 //For each child, call the GetXML function (this will ensure that all children are added recursively)
  370.                 foreach($this->tagChildren as $child)
  371.                     $out .= $child->GetXML();
  372.  
  373.                 //Add the newline and indentation to go along with the close tag
  374.                 $out .= "\n".str_repeat("\t", $this->tagParents);
  375.             }
  376.             
  377.             //If there is data, close off the start tag and add the data
  378.             elseif(!empty($this->tagData))
  379.                 $out .= '>'.$this->tagData;
  380.             
  381.             //Add the end tag    
  382.             $out .= '</'.$this->tagName.'>';
  383.         }
  384.         
  385.         //Return the final output
  386.         return $out;
  387.     }

  389.     /**

  391.      * of $childIndex

  393.      * @param string $childName 
  394.      * @param int $childIndex 
  395.      */
  396.     public function Delete($childName, $childIndex = 0)
  397.     {
  398.         //Delete all of the children of that child
  399.         $this->{$childName}[$childIndex]->DeleteChildren();
  400.         
  401.         //Destroy the child's value
  402.         $this->{$childName}[$childIndex] = null;
  403.         
  404.         //Remove the child's name from the named array
  405.         unset($this->{$childName}[$childIndex]);
  406.         
  407.         //Loop through the tagChildren array and remove any null
  408.         //values left behind from the above operation
  409.         for($x = 0; $x < count($this->tagChildren); $x ++)
  410.         {
  411.             if(is_null($this->tagChildren[$x]))
  412.                 unset($this->tagChildren[$x]);
  413.         }
  414.     }

  417.      * Removes all of the children of this tag in both name and value

  419.     private function DeleteChildren()
  420.     {
  421.         //Loop through all child tags
  422.         for($x = 0; $x < count($this->tagChildren); $x ++)

  424.             //Do this recursively
  425.             $this->tagChildren[$x]->DeleteChildren();
  426.             
  427.             //Delete the name and value
  428.             $this->tagChildren[$x] = null;
  429.             unset($this->tagChildren[$x]);
  430.         }
  431.     }
